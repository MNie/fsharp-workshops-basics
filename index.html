let rec partitionEvenOdd even odd numbers =
    match numbers with
    | [] ->
        (even, odd)
    | h :: tail when h % 2 = 0 ->
        partitionEvenOdd (h :: even) odd tail
    | h :: tail when h % 2 = 1 ->
        partitionEvenOdd even (h :: odd) tail

let ``example 4.1`` = partitionEvenOdd [] [] [1..10]
